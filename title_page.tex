% ---------- ORCID 快捷命令 ----------
% 用法：\orcid{0000-0002-1825-0097}
\newcommand{\orcid}[1]{\orcidlink{#1}}

% ============================================================
%  标题
%  要求：简洁、信息充分；避免缩写和公式（公认的如 DNA 除外）
% ============================================================
\newcommand{\papertitle}{%
  Your Article Title Here%                         <-- 替换为你的标题
}

% ============================================================
%  作者条目（无脚注版本）
%  用法：\authoritem[头衔]{姓名全称}{单位字母}{ORCID}{标记}
%
%  参数说明：
%    #1 = （可选）头衔，如 Prof.、Dr.，不需要则留空 []
%    #2 = 姓名全称。建议 "Ming Li (李铭)"，姓和名不拆开
%    #3 = 单位上标字母，如 a 或 a,b（与下方 \affiliation 对应）
%    #4 = ORCID（16 位数字），不需要则留空 {}
%    #5 = 额外标记（仅作上标显示，不产生脚注）：
%         *   = 通讯作者（页面下方统一声明）
%         \S  = 现址/永久地址（页面下方统一声明）
%
%  示例：
%    \authoritem[Prof.]{Ming Li (李铭)}{a}{0000-0001-2345-6789}{*}
%    \authoritem{Wei Zhang}{a}{}{}
%    \authoritem[Dr.]{John Smith}{b}{0000-0003-4567-8901}{\S}
% ============================================================
\newcommand{\authoritem}[5][]{%
  \if\relax\detokenize{#1}\relax\else#1\,\fi
  \textbf{#2}%
  \if\relax\detokenize{#3}\relax\else\textsuperscript{#3}\fi
  \if\relax\detokenize{#5}\relax\else\textsuperscript{,#5}\fi
  \if\relax\detokenize{#4}\relax\else\ \orcid{#4}\fi
}

% ---------- 内部存储命令 ----------
% 以下命令仅存储信息，不产生任何输出，供页面下方统一显示
\newcommand{\tpcorrname}{}
\newcommand{\tpcorraffil}{}
\newcommand{\tpcorremail}{}
\newcommand{\tpcorrphone}{}
\newcommand{\corrauthor}[4]{%
  \renewcommand{\tpcorrname}{#1}%
  \renewcommand{\tpcorraffil}{#2}%
  \renewcommand{\tpcorremail}{#3}%
  \renewcommand{\tpcorrphone}{#4}%
}

\newcommand{\tppresentaddr}{}
\newcommand{\presentaddress}[1]{\renewcommand{\tppresentaddr}{#1}}

% ============================================================
%  单位条目
%  用法：\affiliation{字母}{完整地址}
%
%  要求：
%   - 完整邮寄地址，必须包含国家名
%   - 上标字母必须与 \authoritem 中的第 2 个参数（单位字母）一致
%
%  示例：
%    \affiliation{a}{School of Materials Science and Engineering,
%                    Tsinghua University, Beijing 100084, China}
%    \affiliation{b}{Department of Engineering Science,
%                    University of Oxford, Oxford OX1 3PJ, UK}
% ============================================================
\newcommand{\affiliation}[2]{%
  \textsuperscript{#1}\,#2\par\vspace{2pt}%
}

% ============================================================
%  组装文档
% ============================================================

\begin{document}

\vspace{0.8cm}

% ---- 作者列表（左对齐，正文字号） ----
\begin{flushleft}
  % 作者 1：通讯作者（可选头衔：Prof. / Dr. 等，不需要则删掉 [Prof.]）
  \authoritem[Prof.]{San Zhang (张三)}{a}{0000-0001-2345-6789}{*}% <-- 替换
  \corrauthor%
    {San Zhang (张三)}%                              <-- 通讯作者姓名
    {Department/School, University Name, City, Country}%  <-- 通讯作者单位
    {san.zhang@example.edu.cn}%                      <-- 通讯作者邮箱
    {+86-10-12345678}%                               <-- 联系电话
  \\[4pt]
  % 作者 2：同单位，无 ORCID，无头衔
  \authoritem{Si Li (李四)}{a}{}{}%                          <-- 替换
  \\[4pt]
  % 作者 3：不同单位，有 ORCID，已调离
  \authoritem[Dr.]{Wu Wang (王五)}{b}{0000-0003-4567-8901}{\S}%    <-- 替换
  \presentaddress{Department of Materials Science and Engineering,
                  MIT, Cambridge, MA 02139, USA}%            <-- 替换/删除
  \\[4pt]
  % 作者 4：归属两个单位
  \authoritem{Liu Zhao (赵六)}{a,b}{0000-0004-5678-9012}{}%  <-- 替换
\end{flushleft}
